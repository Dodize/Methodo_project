-- Ce code intermédiaire calcule la factorielle F d’un entier n
-- avec numéros de ligne
Programme Facto est
    n, i, Fact, T1, T2, T3 : Entier
Début
n <- 5
i <- 1
Fact <- 1
T1 <- i < n
T2 <- i = n
T3 <- T1 OR T2
IF T3 GOTO 9
GOTO 15
Fact <- Fact * i
i <- i + 1
T1 <- i < n
T2 <- i = n
T3 <- T1 OR T2
GOTO 7
NULL
Fin
